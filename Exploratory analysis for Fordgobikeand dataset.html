#Get the distance of all trips accourding to latitude and langitude of start and end stations of trips
from math import sin, cos, sqrt, atan2,radians
R = 6373.0

#we need to convert latitude to radian before make calculations
lat1 = trips['start_station_latitude'].apply(lambda x: radians(x))   
lon1 = trips['start_station_longitude'].apply(lambda x: radians(x))
lat2 = trips['end_station_latitude'].apply(lambda x: radians(x))
lon2 = trips['end_station_longitude'].apply(lambda x: radians(x))

#dlat is the difference between latitude of start station and end station
dlat = lat2 - lat1

#dlon is the difference between longitude of start station and end station
dlon = lon2 - lon1

#finally get the distance through following equations
a = dlat.apply(lambda x: sin(x / 2)**2) + lat1.apply(lambda x: cos(x)) * lat2.apply(lambda x: cos(x)) * dlon.apply(lambda x: sin(x / 2)**2)
c = 2 * a.apply(lambda x: atan2(sqrt(x), sqrt(1 - x)))

trips['distance'] = R * c
